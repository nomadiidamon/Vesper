41 {
42 private:
43 InstrumentationSession* m_CurrentSession;
44 std::ofstream m_OutputStream;

46 public:
47 Instrumentor()
48 : m_CurrentSession(nullptr)

52 void BeginSession(const std::string& name, const std::string& filepath = "results.json")
53 {
54 std::lock_guard lock(m_Mutex);
55 if (m_CurrentSession) {
56 // If there is already a current session, end it and start new one
57 if (Log::GetCoreLogger())
58 {
59 VZ_CORE_ERROR("Instrumentor::BeginSession('{0}') when session '{1}' already open.", name, m_CurrentSession->Name);
60 }
61 InternalEndSession();
62 }
63 m_OutputStream.open(filepath);
64 if (m_OutputStream.is_open()) {
65 m_CurrentSession = new InstrumentationSession{ name };
66 WriteHeader();
67 }
68 else {
69 if (Log::GetCoreLogger())
70 {
71 VZ_CORE_ERROR("Instrumentor could not open results file '{0}'.", filepath);
72 }
73 }
74 WriteHeader();
75 m_CurrentSession = new InstrumentationSession{ name };
76 }
77
78 void EndSession()
79 {
80 std::lock_guard lock(m_Mutex);
81 InternalEndSession();
82 }

84 void WriteProfile(const ProfileResult& result)
85 {
86 m_OutputStream << ",";
87
88 std::string name = result.Name;
89 std::replace(name.begin(), name.end(), '"', '\'');
90
91 m_OutputStream << "{";
92 m_OutputStream << "\"cat\":\"function\",";
93 m_OutputStream << "\"dur\":" << (result.End - result.Start) << ',';
94 m_OutputStream << "\"name\":\"" << name << "\",";
95 m_OutputStream << "\"ph\":\"X\",";
96 m_OutputStream << "\"pid\":0,";
97 m_OutputStream << "\"tid\":" << result.ThreadID << ",";
98 m_OutputStream << "\"ts\":" << result.Start;
99 m_OutputStream << "}";
100
101 std::lock_guard lock(m_Mutex);
102 if (m_CurrentSession)
103 {
104 //m_OutputStream << json.str();
105 m_OutputStream.flush();
106 }
107 }
108
109 void WriteHeader()
110 {
111 m_OutputStream << "{\"otherData\": {},\"traceEvents\":[";
112 m_OutputStream.flush();
113 }
114
115 void WriteFooter()
116 {
117 m_OutputStream << "]}";
118 m_OutputStream.flush();
119 }
120
121 void InternalEndSession()
122 {
123 if (m_CurrentSession)
124 {
125 WriteFooter();
126 m_OutputStream.close();
127 delete m_CurrentSession;
128 m_CurrentSession = nullptr;
129 }
130 }

139 class InstrumentationTimer
140 {
141 public:
142 InstrumentationTimer(const char* name)
143 : m_Name(name), m_Stopped(false)
144 {
145 m_StartTimepoint = std::chrono::high_resolution_clock::now();
146 }
147
148 ~InstrumentationTimer()
149 {
150 if (!m_Stopped)
151 Stop();
152 }
153
154 void Stop()
155 {
156 auto endTimepoint = std::chrono::high_resolution_clock::now();
157
158 long long start = std::chrono::time_point_cast<std::chrono::microseconds>(m_StartTimepoint).time_since_epoch().count();
159 long long end = std::chrono::time_point_cast<std::chrono::microseconds>(endTimepoint).time_since_epoch().count();
160
161 uint32_t threadID = std::hash<std::thread::id>{}(std::this_thread::get_id());
162 Instrumentor::Get().WriteProfile({ m_Name, start, end, threadID });
163
164 m_Stopped = true;
165 }
166 private:
167 const char* m_Name;
168 std::chrono::time_point<std::chrono::high_resolution_clock> m_StartTimepoint;
169 bool m_Stopped;
170 };

177 template <size_t N>
178 struct ChangeResult
179 {
180 char Data[N];
181 };
182
183 template <size_t N, size_t K>
184 constexpr auto CleanupOutputString(const char(&expr)[N], const char(&remove)[K])
185 {
186 ChangeResult<N> result = {};
187
188 size_t srcIndex = 0;
189 size_t dstIndex = 0;
190 while (srcIndex < N)
191 {
192 size_t matchIndex = 0;
193 while (matchIndex < K - 1 && srcIndex + matchIndex < N - 1 && expr[srcIndex + matchIndex] == remove[matchIndex])
194 matchIndex++;
195 if (matchIndex == K - 1)
196 srcIndex += matchIndex;
197 result.Data[dstIndex++] = expr[srcIndex] == '"' ? '\'' : expr[srcIndex];
198 srcIndex++;
199 }
200 return result;
201 }
202}
