14 /// @brief Returns the size in bytes of the given ShaderDataType.

95 CalculateOffsetsAndStride();
96 }

99 /// @brief Returns the list of BufferElements in the layout.
100 inline const std::vector<BufferElement>& GetElements() const { return m_Elements; }
101 /// @brief Returns the stride (total size in bytes) of the buffer layout.
102 inline uint32_t GetStride() const { return m_Stride; }
103
104 std::vector<BufferElement>::iterator begin() { return m_Elements.begin(); }
105 std::vector<BufferElement>::const_iterator begin() const { return m_Elements.begin(); }
106 std::vector<BufferElement>::iterator end() { return m_Elements.end(); }
107 std::vector<BufferElement>::const_iterator end() const { return m_Elements.end(); }
108
109
110
111 private:
112 /// @brief Calculates the offsets and stride for the buffer layout based on its elements.
113 void CalculateOffsetsAndStride() {
114 uint32_t offset = 0;
115 m_Stride = 0;
116 for (auto& element : m_Elements) {
117 element.Offset = offset;
118 offset += element.Size;
119 m_Stride += element.Size;
120 }
121 }
